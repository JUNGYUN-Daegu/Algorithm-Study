```swift
var dict = [Character: Int]()
var containZero = false
let nums = readLine()!.map{ c -> Int in
    if c == "0" {
        containZero = true
    }
    return Int(String(c))!
}

if !containZero || nums.reduce(0, +) % 3 != 0 {
    print(-1)
} else {
    print(nums.map{ String($0) }.sorted(by: >).joined())
}
```
